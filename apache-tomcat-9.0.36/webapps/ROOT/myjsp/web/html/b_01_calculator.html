< script type="text/javascript">
    var show="";
    function TNumber(num) {
        show += num;
        document.getElementById("txt").value=show;
    }


    function Calculator() {
        var result = show;
        Clear();
        document.getElementById("txt").value=result+"="+eval(result);
        show = eval(result).toString();
    }


    function Clear() {
        show = "";
